    # _init

    // src/core/instance/init.js
function initMixin (Vue) {
    // 就是 new Vue() 主要干的事情
    Vue.prototype._init = function (options) {
        // 保存当前实例到 vm
        const vm: Component = this
        // 每一个实例上都有一个 _uid, 每一个都不一样
        vm._uid = uid++
        
        // 作为一个标记，避免被重复的 observe
        vm._isVue = true
        
        // 判断初始化的是否是组件，并且存在 options 选项
        if (options && options._isComponent) {
            initInternalComponent(vm, options)
        } else {
            // 合并属性
            vm.$options = mergeOptions(
                resolveConstructorOptions(vm.constructor),
                options || {},
                vm
            )
        }

        
        // 初始化组件生命周期标志位
        initLifecycle(vm)
        // 初始化组件事件侦听
        initEvents(vm)
        // 初始化渲染方法
        initRender(vm)
        // 调用生命周期钩子函数 第一个
        callHook(vm, 'beforeCreate')
        // 初始化依赖注入内容，在初始化data、props 之前。
        initInjections(vm) // resolve injections before data/props
        // 初始化 props/data/method/watch/methods
        initState(vm)
        // 初始化 provide
        initProvide(vm) // resolve provide after data/props
        // 调用生命周期钩子函数
        callHook(vm, 'created')

        /* istanbul ignore if */
        if (process.env.NODE_ENV !== 'production' && config.performance && mark) {
        vm._name = formatComponentName(vm, false)
        mark(endTag)
        measure(`vue ${vm._name} init`, startTag, endTag)
        }

        // 初始化好之后。
        // 判断用户是否传入 el 选项,
        // 如果传入了则调用$mount函数进入模板编译与挂载阶段
        // 如果没有传入 el， 不进入下一个生命周期阶段， 需要用户手动调用$mount 方法进入下一个生命周期阶段。
        if (vm.$options.el) {
        // 获取到用户传入的模板内容并将其编译成渲染函数
        vm.$mount(vm.$options.el)
        }
    }
}

    mergeOptions 进行属性的合并

    function mergeOptions (
  parent: Object,
  child: Object,
  vm?: Component
): Object {
  if (process.env.NODE_ENV !== 'production') {
    checkComponents(child)
  }
  // 如果 child 是一个函数的话, 获取 child 上的options 选项
  if (typeof child === 'function') {
    child = child.options
  }

  // Vue 不存在
  if (!child._base) {
    // 首先把 parent 和 child 这两个对象根据一些合并策略，合并成一个新对象并返回。
    // 递归把 extends 传入的数据 合并到 parent 上。
    if (child.extends) {
      parent = mergeOptions(parent, child.extends, vm)
    }
    // 递归把 mixins 传入的数据, 合并到 parent 上.
    if (child.mixins) {
      // 遍历每一个mixins, 合并到 parent 上.
      for (let i = 0, l = child.mixins.length; i < l; i++) {
        parent = mergeOptions(parent, child.mixins[i], vm)
      }
    }
  }

  // 创建一个空对象 options
  const options = {}
  let key
  // 遍历 parent
  for (key in parent) {
    // 通过这个函数合并到空对象 options 里
    mergeField(key)
  }
  // 遍历 child
  for (key in child) {
    // 将存在于 child 里但是不在 parent 中的属性继续调用这个函数合并到空对象 options
    if (!hasOwn(parent, key)) {
      mergeField(key)
    }
  }
  // 设计模式中的策略模式
  // 合并策略是根据被合并的选项有着不同的合并策略。
  // strats.data 是 data的合并策略。
  // watch 有着 watch 的合并策略。
  function mergeField (key) {
    // 获取到 strat = strats[data]
    const strat = strats[key] || defaultStrat
    // 根据不同的合并策略
    // strat(parent[data], child[key])  给到 options[data]
    options[key] = strat(parent[key], child[key], vm, key)
  }
  // 返回这个合并到对象
  return options
}

    # initLifecycle

    function initLifecycle (vm: Component) {
  // 获取到 vm.$options 的选项
  const options = vm.$options

  // locate first non-abstract parent
  let parent = options.parent
  // 如果当前组件不是抽象组件，并且存在父级
  if (parent && !options.abstract) {
    // 向上循环，如果当前组件的父级是抽象组件并且存在父级。
    while (parent.$options.abstract && parent.$parent) {
      // 继续向上查找当前组件父级的父级。
      parent = parent.$parent
    }

    // 将自己的实例添加到父级的 $children 属性中。
    parent.$children.push(vm)
  }
  // 直到找到第一个不是抽象类型的父级.
  //将其赋值给 $parent
  vm.$parent = parent

  // 挂载 $root 属性
  // 判断当前实例存在父级，跟实例就是 其父的跟实例，如果不存在，跟实例就是 它自己。
  vm.$root = parent ? parent.$root : vm
  // 供外部使用
  vm.$children = []
  vm.$refs = {}

  // 初始化一些 _ 的属性, 供内部使用
  vm._watcher = null
  vm._inactive = null
  vm._directInactive = false
  vm._isMounted = false
  vm._isDestroyed = false
  vm._isBeingDestroyed = false
}

function initEvents (vm: Component) {

  // 实例上创建一个 _events 属性的空对象, 用来存储事件
  vm._events = Object.create(null)
  vm._hasHookEvent = false
  // init parent attached events

  // 获取父组件注册的事件赋给 listeners
  const listeners = vm.$options._parentListeners
  // 如果 listeners 不为空
  if (listeners) {
    // 调用这个函数，将父组件向子组件注册的事件注册到子组件的实例中。
    updateComponentListeners(vm, listeners)
  }
}

    在 initEvents 初始化中，会将自定义事件交给子组件处理，而原生事件是父组件来处理的。

    调用 updateComponentListeners 将父组件向子组件注册的事件注册到子组件的实例中。

    // src/core/instance/events.js
function updateComponentListeners (
  vm: Component,
  listeners: Object,
  oldListeners: ?Object
) {
  target = vm

  // 只是调用了一个 updateListeners 这个函数， 传入 add 和 remove 两个函数
  // 如果listeners对象中存在某个key（即事件名）而oldListeners中不存在，则说明这个事件是需要新增的；反之，如果oldListeners对象中存在某个key（即事件名）而listeners中不存在，则说明这个事件是需要从事件系统中卸载的；
  updateListeners(listeners, oldListeners || {}, add, remove, createOnceHandler, vm)
  target = undefined
}

    // src/core/vdom/helpers/update-listeners.js

// 对比 listeners 和 oldListeners 的不同。
// 调用参数中提供的 add 和 remove 进行相应的注册事件和卸载事件。
// 如果 listeners 中存在 而 oldListeners 中不存在就要新增。
// 如果 listeners 中不存在，而 oldListeners 中存在，就要删除。
export function updateListeners (
  on: Object,
  oldOn: Object,
  add: Function,
  remove: Function,
  createOnceHandler: Function,
  vm: Component
) {
  let name, def, cur, old, event

  // 对 on 进行遍历，获得每一个事件名
  for (name in on) {
    def = cur = on[name]
    old = oldOn[name]
    event = normalizeEvent(name)
    /* istanbul ignore if */
    if (__WEEX__ && isPlainObject(def)) {
      cur = def.handler
      event.params = def.params
    }
    // 判断对应的事件名是否在 listeners 中存在
    if (isUndef(cur)) {
      // 不存在就抛出警告
      process.env.NODE_ENV !== 'production' && warn(
        `Invalid handler for event "${event.name}": got ` + String(cur),
        vm
      )
      // 判断是否在 oldListeners 中存在
    } else if (isUndef(old)) {
      // 如果不存在, 看看在 listeners.fns 是不是存在
      if (isUndef(cur.fns)) {
        // 执行完之后返回 invoker, 在 cur 上面添加 fns
        cur = on[name] = createFnInvoker(cur, vm)
      }
      if (isTrue(event.once)) {
        cur = on[name] = createOnceHandler(event.name, cur, event.capture)
      }
      // 如果不在 oldListeners 中存在, 就 add 注册
      add(event.name, cur, event.capture, event.passive, event.params)

      // 如果新旧不同
    } else if (cur !== old) {
      // 之前绑定的 involer.fns 赋值为新的回调函数即可
      old.fns = cur
      // 通过 on[name] = old 保留引用关系，这样就保证了事件回调只添加一次，之后仅仅去修改它的回调函数的引用。
      on[name] = old
    }
  }

  // 最后遍历 oldOn
  for (name in oldOn) {
    //  获得每一个事件名，判断如果事件名在on中不存在
    if (isUndef(on[name])) {
      event = normalizeEvent(name)
      // 则表示该事件是需要从事件系统中卸载的事件，则调用 remove方法卸载该事件。
      remove(event.name, oldOn[name], event.capture)
    }
  }
}

    mergeHook

    function mergeHook (
  parentVal: ?Array<Function>,
  childVal: ?Function | ?Array<Function>
): ?Array<Function> {
  const res;

  if (childVal) {
    if (parentVal) {
      res = parentVal.concat(childVal);
    } else {
      if(Array.isArray(childVal)) {
        res = childVal;
      } else {
        res = [childVal];
      }
    }
  } else {
    res = parentVal;
  }

  // 返回这个数组
  return res
    ? dedupeHooks(res)
    : res
}

function dedupeHooks (hooks) {
  const res = []
  // 遍历每一个钩子
  for (let i = 0; i < hooks.length; i++) {
    // 做一个去重操作, 没有找到就去添加
    if (res.indexOf(hooks[i]) === -1) {
      res.push(hooks[i])
    }
  }
  return res
}

    // src/core/instance/lifecycle.js

// 触发钩子函数。
export function callHook (vm: Component, hook: string) {
  // #7573 disable dep collection when invoking lifecycle hooks
  // 全局的 Watcher
  pushTarget()
  // 从 $options 中获取到需要触发的钩子名称所对应的数组。
  const handlers = vm.$options[hook]
  const info = `${hook} hook`
  // 如果存在就去遍历
  if (handlers) {
    for (let i = 0, j = handlers.length; i < j; i++) {
      // 将数组中的每一个钩子函数都执行一遍
      invokeWithErrorHandling(handlers[i], vm, null, vm, info)
    }
  }
  if (vm._hasHookEvent) {
    vm.$emit('hook:' + hook)
  }
  popTarget()
}
